# SP1: W4A4 NVFP4 GEMM op — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** sm_120(RTX PRO 6000)에서 수치 정확하고 벤치마크된, Python 호출 가능한 dense W4A4 NVFP4 GEMM op을 만든다.

**Architecture:** 디바이스 무관 양자화/패킹/역양자화 유틸 + fp32 "emulated" 참조 오라클을 먼저 TDD로 세우고, 그 위에 균일한 `nvfp4_gemm`/`nvfp4_linear` op 인터페이스를 둔다. 실제 텐서코어 백엔드는 **재사용 우선 사다리**(FlashInfer b12x → CUTLASS-79+정렬픽스 → cuBLASLt → 직접 빌드)로 붙이되, 각 백엔드를 동일한 정합성 게이트(참조 대비 cosine/상대오차, all-zeros 검출)로 on-device 판정한다. 통과하는 첫 백엔드가 기본값.

**Tech Stack:** Python 3.12 + uv, PyTorch **cu130**, CUDA 13.1/nvcc 13.1, FlashInfer, CUTLASS 4.x, (조건부) cuBLASLt C++ extension, pytest.

## Global Constraints

- 빌드 타깃 GPU: **sm_120 / compute capability (12, 0)**. 모든 커널 정합성/벤치는 **실제 RTX PRO 6000 위에서** 실행(에뮬레이션 불가).
- PyTorch는 **cu130 휠**(사용자 확정). 환경변수 **`TORCH_CUDA_ARCH_LIST=12.0a`** 를 모든 CUDA 확장 빌드/실행에 설정('a' 누락 시 NVFP4 block-scaled MMA가 조용히 깨짐 — pytorch #172807).
- NVFP4 포맷: E2M1 원소(크기 `{0,0.5,1,1.5,2,3,4,6}`, max **6.0**), 블록 **16**, 블록 스케일 **FP8 E4M3**(UE4M3, max **448.0**), 텐서당 **FP32 global scale**. 2단계 디코드: `x ≈ e2m1 * block_e4m3 * global`.
- 패킹: E2M1 2개/byte, `torch.float4_e2m1fn_x2`. contraction 차원 **K는 16의 배수**(아니면 패딩).
- **SF swizzle(128×4)는 각 백엔드 adapter가 담당**한다. 공유 양자화 유틸은 논리적(unswizzled) 블록 스케일 `(..., K/16)`을 반환한다.
- 패키지: `src/vit_nvfp4/` 레이아웃(uv editable install). 커밋은 task 단위로 자주.

```python
import torch

E2M1_MAX = 6.0
E4M3_MAX = 448.0

# magnitude per 3-bit index
_MAG = torch.tensor([0.0, 0.5, 1.0, 1.5, 2.0, 3.0, 4.0, 6.0], dtype=torch.float32)
# midpoints between consecutive magnitudes (nearest-rounding boundaries)
_MID = torch.tensor([0.25, 0.75, 1.25, 1.75, 2.5, 3.5, 5.0], dtype=torch.float32)

def round_to_e2m1_code(scaled: torch.Tensor) -> torch.Tensor:
    sign = (scaled < 0).to(torch.uint8)
    mag = scaled.abs().clamp(max=E2M1_MAX)
    idx = torch.bucketize(mag, _MID.to(mag.device)).to(torch.uint8)  # 0..7
    return (sign << 3) | idx

def code_to_value(code: torch.Tensor) -> torch.Tensor:
    code = code.to(torch.long)
    sign = 1.0 - 2.0 * ((code >> 3) & 1).to(torch.float32)
    mag = _MAG.to(code.device)[code & 0x7]
    return sign * mag
```

```python
import torch
from . import format as fmt

def quantize_to_nvfp4(x: torch.Tensor, block: int = 16, global_scale: torch.Tensor | None = None):
    assert x.shape[-1] % block == 0, "K must be a multiple of block; pad first"
    xf = x.to(torch.float32)
    if global_scale is None:
        amax = xf.abs().amax().clamp(min=1e-12)
        global_scale = (amax / (fmt.E2M1_MAX * fmt.E4M3_MAX)).to(torch.float32)
    s_enc = (1.0 / global_scale)
    xb = xf.reshape(*xf.shape[:-1], xf.shape[-1] // block, block)
    amax_b = xb.abs().amax(dim=-1, keepdim=True).clamp(min=1e-12)         # (...,K/blk,1)
    block_scale = ((amax_b / fmt.E2M1_MAX) * s_enc).clamp(max=fmt.E4M3_MAX)
    block_scale_e4m3 = block_scale.to(torch.float8_e4m3fn)
    decoded = block_scale_e4m3.to(torch.float32) * global_scale          # ≈ amax_b/6
    codes = fmt.round_to_e2m1_code(xb / decoded).reshape(x.shape)        # uint8 (...,K)
    return codes, block_scale_e4m3.squeeze(-1), global_scale

def dequantize_nvfp4(codes: torch.Tensor, block_scale_e4m3: torch.Tensor, global_scale: torch.Tensor):
    block = codes.shape[-1] // block_scale_e4m3.shape[-1]
    vals = fmt.code_to_value(codes).reshape(*codes.shape[:-1], block_scale_e4m3.shape[-1], block)
    scale = (block_scale_e4m3.to(torch.float32) * global_scale).unsqueeze(-1)
    return (vals * scale).reshape(codes.shape)
```

```python
import torch

def pack_e2m1(codes: torch.Tensor) -> torch.Tensor:
    assert codes.shape[-1] % 2 == 0 and codes.dtype == torch.uint8
    lo = codes[..., 0::2]
    hi = codes[..., 1::2]
    return ((hi << 4) | lo).contiguous()

def unpack_e2m1(packed: torch.Tensor) -> torch.Tensor:
    lo = packed & 0x0F
    hi = (packed >> 4) & 0x0F
    out = torch.stack([lo, hi], dim=-1).reshape(*packed.shape[:-1], packed.shape[-1] * 2)
    return out.to(torch.uint8)

def pad_to_block(x: torch.Tensor, block: int = 16, dim: int = -1):
    n = x.shape[dim]
    pad = (-n) % block
    orig = n
    if pad:
        pad_cfg = [0, 0] * (x.ndim)
        # F.pad pads last dim first; build for arbitrary dim via movedim
        x = x.movedim(dim, -1)
        x = torch.nn.functional.pad(x, (0, pad))
        x = x.movedim(-1, dim)
    return x, orig

def as_float4_x2(packed: torch.Tensor) -> torch.Tensor:
    return packed.view(torch.float4_e2m1fn_x2)
```

```python
import torch, pytest
from vit_nvfp4.nvfp4.quant import quantize_to_nvfp4, dequantize_nvfp4
from vit_nvfp4.nvfp4 import gemm as G

def _make_operands(M, K, N, dist, dtype=torch.float32, dev="cpu"):
    g = torch.Generator(device=dev).manual_seed(1234)
    a = torch.randn(M, K, generator=g, device=dev, dtype=dtype)
    b = torch.randn(N, K, generator=g, device=dev, dtype=dtype)
    if dist == "outlier":
        a[:2] *= 30.0
    a_codes, a_bs, a_gs = quantize_to_nvfp4(a, 16)
    b_codes, b_bs, b_gs = quantize_to_nvfp4(b, 16)
    return (a_codes, a_bs, a_gs), (b_codes, b_bs, b_gs)

def _fp4_emulated_ref(aq, bq):
    a = dequantize_nvfp4(*aq); b = dequantize_nvfp4(*bq)
    return a @ b.T

@pytest.mark.parametrize("M,K,N", [(64, 256, 64), (128, 1024, 256)])
def test_reference_matches_fp4_emulation(M, K, N):
```

```python
import torch
import torch.nn.functional as F
from .quant import quantize_to_nvfp4, dequantize_nvfp4
from .gemm import nvfp4_gemm

def assert_gemm_correct(backend, M, K, N, *, dist="normal", device="cuda",
                        cos_emul=0.999, cos_bf16=0.99, seed=1234):
    g = torch.Generator(device=device).manual_seed(seed)
    a = torch.randn(M, K, generator=g, device=device)
    b = torch.randn(N, K, generator=g, device=device)
    if dist == "outlier":
        a[:2] *= 30.0
    elif dist == "heavy":
        a *= torch.empty(M, K, device=device).exponential_(0.5, generator=g)
    aq = quantize_to_nvfp4(a, 16); bq = quantize_to_nvfp4(b, 16)
    out = nvfp4_gemm(*aq, *bq, out_dtype=torch.float32, backend=backend)
    all_zero = bool((out == 0).all())
    emul = dequantize_nvfp4(*aq) @ dequantize_nvfp4(*bq).T
    bf16 = (a.bfloat16() @ b.bfloat16().T).float()
    m = {
        "all_zero": all_zero,
        "cos_emul": float(F.cosine_similarity(out.flatten(), emul.flatten(), dim=0)),
        "cos_bf16": float(F.cosine_similarity(out.flatten(), bf16.flatten(), dim=0)),
        "rel_emul": float((out - emul).norm() / emul.norm().clamp(min=1e-9)),
    }
    assert not m["all_zero"], f"{backend} returned all zeros (cf. flashinfer #2577)"
    assert m["cos_emul"] >= cos_emul, m
    assert m["cos_bf16"] >= cos_bf16, m
    return m
```

```python
import torch
from flashinfer.gemm import mm_fp4
from flashinfer import SfLayout  # 128x4 swizzle layout
from ..pack import pack_e2m1, as_float4_x2

def _prep(codes, bs):
    packed = as_float4_x2(pack_e2m1(codes))           # (M, K//2) -> float4_e2m1fn_x2
    sf = bs.to(torch.float8_e4m3fn).contiguous()      # logical (M, K//16)
    return packed, sf

def gemm(a_codes, a_bs, a_gs, b_codes, b_bs, b_gs, out_dtype=torch.bfloat16):
    a_p, a_sf = _prep(a_codes, a_bs)
    b_p, b_sf = _prep(b_codes, b_bs)
    # a_descale/b_descale: per-block E4M3 (swizzled by flashinfer) * fp32 global
    return mm_fp4(a_p, b_p.T, a_sf, b_sf, a_gs, b_gs,
                  out_dtype=out_dtype, block_size=16,
                  sf_layout=SfLayout.layout_128x4, backend="b12x")
```

```python
import torch, time, itertools
from vit_nvfp4.nvfp4.quant import quantize_to_nvfp4
from vit_nvfp4.nvfp4.gemm import nvfp4_gemm, _default_backend

SHAPES = [(M,K,N) for M in (256,1024,4096)
          for (K,N) in ((1024,1024),(1152,4304),(1408,6144))]

def _t(fn, iters=50):
    fn(); torch.cuda.synchronize()
    s = time.perf_counter()
    for _ in range(iters): fn()
    torch.cuda.synchronize()
    return (time.perf_counter()-s)/iters

def bench():
    be = _default_backend()
    for M,K,N in SHAPES:
        a = torch.randn(M,K,device="cuda"); b = torch.randn(N,K,device="cuda")
        aq = quantize_to_nvfp4(a,16); bq = quantize_to_nvfp4(b,16)
        ab, bb = a.bfloat16(), b.bfloat16()
        flops = 2*M*N*K
        t_fp4 = _t(lambda: nvfp4_gemm(*aq,*bq,out_dtype=torch.bfloat16,backend=be))
        t_bf16 = _t(lambda: ab @ bb.T)
        af = a.to(torch.float8_e4m3fn); bf = b.to(torch.float8_e4m3fn)
        sa = torch.ones(1,device="cuda"); 
        t_fp8 = _t(lambda: torch._scaled_mm(af, bf.T, scale_a=sa, scale_b=sa, out_dtype=torch.bfloat16))
        print(f"{M}x{K}x{N}: fp4={flops/t_fp4/1e12:.1f} bf16={flops/t_bf16/1e12:.1f} "
              f"fp8={flops/t_fp8/1e12:.1f} TFLOP/s | fp4/bf16={t_bf16/t_fp4:.2f}x")

if __name__ == "__main__":
    bench()
```
